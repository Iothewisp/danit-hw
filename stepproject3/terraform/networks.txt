# 🏗 Создание VPC
resource "aws_vpc" "vpc_1webseeker" {
  cidr_block           = "10.0.0.0/16"
  enable_dns_support   = true
  enable_dns_hostnames = true

  tags = {
    Name = "webseekersVPC"
  }
}

# 🌐 Публичная Subnet
resource "aws_subnet" "public_subnet" {
  vpc_id                  = aws_vpc.vpc_1webseeker.id
  cidr_block              = "10.0.1.0/24"
  map_public_ip_on_launch = true  

  tags = {
    Name = "webseekers-public-subnet"
  }
}

# 🔒 Приватная Subnet (для Jenkins Worker)
resource "aws_subnet" "private_subnet" {
  vpc_id     = aws_vpc.vpc_1webseeker.id
  cidr_block = "10.0.2.0/24"

  tags = {
    Name = "webseekers-private-subnet"
  }
}

# 🚀 Интернет-шлюз (IGW)
resource "aws_internet_gateway" "igw" {
  vpc_id = aws_vpc.vpc_1webseeker.id

  tags = {
    Name = "webseekervpc-igw"
  }
}

# 🌍 Elastic IP для NAT Gateway
resource "aws_eip" "nat_eip_webseeker" {
  domain = "vpc"

  tags = {
    Name = "webseekers-nat-eip"
  }
}

# 🔁 NAT Gateway для приватной подсети
resource "aws_nat_gateway" "nattty" {
  allocation_id = aws_eip.nat_eip_webseeker.id
  subnet_id     = aws_subnet.public_subnet.id

  tags = {
    Name = "webseekers-nat-gateway"
  }
}

# 🛣 Таблица маршрутов для публичной подсети
resource "aws_route_table" "public_rt" {
  vpc_id = aws_vpc.vpc_1webseeker.id

  tags = {
    Name = "webseekervpc-public-route-table"
  }
}

# 🚦 Публичный маршрут в интернет
resource "aws_route" "default_route" {
  route_table_id         = aws_route_table.public_rt.id
  destination_cidr_block = "0.0.0.0/0"
  gateway_id             = aws_internet_gateway.igw.id
}

# 🔗 Ассоциация публичной Route Table с публичной Subnet
resource "aws_route_table_association" "public_assoc" {
  subnet_id      = aws_subnet.public_subnet.id
  route_table_id = aws_route_table.public_rt.id
}

# 🛣 Таблица маршрутов для приватной подсети
resource "aws_route_table" "private_rt" {
  vpc_id = aws_vpc.vpc_1webseeker.id

  tags = {
    Name = "webseekers-private-route-table"
  }
}

# 🚦 Маршрут из приватной подсети через NAT
resource "aws_route" "private_to_nat" {
  route_table_id         = aws_route_table.private_rt.id
  destination_cidr_block = "0.0.0.0/0"
  nat_gateway_id         = aws_nat_gateway.nattty.id
}

# 🔗 Ассоциация приватной Route Table с приватной Subnet
resource "aws_route_table_association" "private_assoc" {
  subnet_id      = aws_subnet.private_subnet.id
  route_table_id = aws_route_table.private_rt.id
}
